INCLUDE(UsePkgConfig)

# Include our custom cmake modules
SET(CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/CMake)

FIND_PACKAGE(Qt4 REQUIRED)
FIND_PACKAGE(OpenAL REQUIRED)
FIND_PACKAGE(Boost REQUIRED)
FIND_PACKAGE(OggVorbis REQUIRED)

# Fail if Qt doesn't have OpenGL support
IF (NOT QT_QTOPENGL_FOUND)
	MESSAGE(FATAL_ERROR "Qt OpenGL libraries were not found.")
ENDIF (NOT QT_QTOPENGL_FOUND)

# Find freetype
PKGCONFIG(freetype2 FREETYPE_INCLUDE_DIR FREETYPE_LIB_DIR FREETYPE_LINK_FLAGS FREETYPE_CFLAGS)

IF (NOT FREETYPE_INCLUDE_DIR)
	MESSAGE(FATAL_ERROR "Freetype was not found.")
ENDIF (NOT FREETYPE_INCLUDE_DIR)


# Set up definitions and paths
ADD_DEFINITIONS(${QT_DEFINITIONS})
LINK_DIRECTORIES(${QT_LIBRARY_DIR} ${FREETYPE_LIB_DIR} ${Boost_LIBRARY_DIRS})
INCLUDE_DIRECTORIES(${QT_INCLUDES} ${FREETYPE_INCLUDE_DIR} ${OPENAL_INCLUDE_DIR} ${Boost_INCLUDE_DIRS} ${OGGVORBIS_INCLUDE_DIR})

# Subdirectories
ADD_SUBDIRECTORY(src)


# Uninstall support
CONFIGURE_FILE(
  "${CMAKE_CURRENT_SOURCE_DIR}/cmake_uninstall.cmake.in"
  "${CMAKE_CURRENT_BINARY_DIR}/cmake_uninstall.cmake"
  IMMEDIATE @ONLY)

ADD_CUSTOM_TARGET(uninstall
  "${CMAKE_COMMAND}" -P "${CMAKE_CURRENT_BINARY_DIR}/cmake_uninstall.cmake")
